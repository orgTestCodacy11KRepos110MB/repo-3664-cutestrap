This example demonstrates listening for a button click. When the button is clicked, the content is updated to say 'clicked', instead of the original 'Click Me'

Click Me 
<button>Click Me</button>

<script type="module">
  import {
    get,
    set,
    on,
    load
  } from '../js/vendor/ion/app.js?CACHEBUST'

  const namespace = 'super-button'

  const initial = { clicks: 0 }
  const count = () => (get(namespace) || initial).clicks

  function add() {
    const clicks = count() + 1
    set(namespace, { clicks })
  }

  function render({target}) {
    target.innerHTML = `
      <button>
        ${count() || target.innerText}
      </button>
    `
  }

  on('click', `${namespace} button`, add)
  on('render', namespace, render)

  load(namespace)
</script>
